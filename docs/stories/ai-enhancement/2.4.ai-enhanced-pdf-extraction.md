### Integration Approach

**Hybrid Strategy - Best of Both Worlds:**

1. **Traditional Parser First** - Baseline extraction usando parser atual
2. **AI Validation Layer** - Valida cada questão extraída com LLama3
3. **Selective Repair** - Só usa IA para questões com problemas identificados
4. **Missing Detection** - IA busca questões perdidas em chunks não processados

### Technical Architecture

```python
# Nova arquitetura híbrida
class AIEnhancedEnemParser:
    def __init__(self):
        self.traditional_parser = EnemIngestionParser()  # Parser atual
        self.ai_validator = QuestionValidationService()
        self.ai_repairer = QuestionRepairService()
        self.missing_detector = MissingQuestionDetector()
        self.metrics_tracker = ExtractionMetricsTracker()
    
    async def extract_questions_hybrid(self, pdf_path: str) -> ExtractionResult:
        # Phase 1: Traditional extraction (baseline)
        traditional_result = self.traditional_parser.extract_questions(pdf_path)
        
        # Phase 2: AI validation of extracted questions
        validated_questions = []
        for question in traditional_result.questions:
            validation = await self.ai_validator.validate_question(question)
            
            if validation.confidence > 0.8:
                validated_questions.append(question)
            elif validation.confidence > 0.4:
                # Try to repair low-confidence questions
                repaired = await self.ai_repairer.repair_question(
                    question, validation.issues
                )
                validated_questions.append(repaired)
            # Questions with confidence < 0.4 go to manual review queue
        
        # Phase 3: Detect missing questions using AI
        missing_questions = await self.missing_detector.find_missing_questions(
            pdf_path, validated_questions
        )
        
        # Phase 4: Combine results and generate metrics
        final_result = ExtractionResult(
            questions=validated_questions + missing_questions,
            metrics=self.metrics_tracker.calculate_metrics(
                traditional_result, validated_questions, missing_questions
            )
        )
        
        return final_result
```

### AI Processing Optimizations

```python
# Otimizações para performance
class AIProcessingOptimizer:
    def __init__(self):
        self.question_cache = {}  # Cache validações por hash da questão
        self.batch_processor = BatchProcessor(max_batch_size=5)
        self.confidence_threshold = 0.4  # Só processa questões abaixo
    
    async def process_questions_optimized(self, questions: List[Question]):
        # Só valida questões que não estão em cache
        uncached_questions = [q for q in questions if not self._is_cached(q)]
        
        # Processa em batches para efficiency
        results = await self.batch_processor.process_batch(uncached_questions)
        
        # Combina cached + new results
        return self._combine_results(questions, results)
```
